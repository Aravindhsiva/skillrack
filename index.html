import java.util.Scanner;
import java.math.*;
public class midrev {
 static String rev(String s) {
  StringBuilder r = new StringBuilder();
  r.append(s);
  r.reverse();
  return r.toString();
 }
 public static void main(String[] args) {
  Scanner x = new Scanner(System.in);
  String s = x.next();
  int n = x.nextInt();
  String c = s.substring(0,s.length()/2);
  String d = s.substring(s.length()/2,s.length());
  String f = c.substring(c.length()-n/2,c.length());
  String g = d.substring(0,n-(int)Math.floor(n/2));
  String rev = rev(f+g);
  int rem = (int)Math.floor(s.length()/2) - (int)Math.floor(n/2);
  System.out.println(s.substring(0,rem)+rev+s.substring(s.length()-rem,s.length()));
 }
}
