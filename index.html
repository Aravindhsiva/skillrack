#include<stdio.h>
int max(int a[],int n)
{
int i,max;
max=a[0];
for(i=0;i<n;i++)
{
    if(a[i]>max)
    max=a[i];
}
    return max;
}
void main(void)
{
int a[25][25],b[25],m,n,i,j;
scanf("%d%d",&m,&n);
for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
            scanf("%d",&a[i][j]);
}
for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
            b[j]=a[i][j];
            printf("%d\n",max(b,n));
    }
}
