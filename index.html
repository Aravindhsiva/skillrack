import java.util.*;
public class Hello {

    public static void main(String[] args) {
  int n,flag=0;
  Scanner x = new Scanner(System.in);
  n = x.nextInt();
  int a[] = new int[n];
  int cc=0;
  for(int i=0;i<n;i++)
      a[i] = x.nextInt();
  if(a.length>1){
  for(int i=0;i<n;i++){
      for(int j=0;j<n;j++){
          if(i!=j){
              if(a[i]!=a[j])
                  flag=1;
              else{
                  flag=0;
                  break;
              }
          }
      }
      if(flag==1)
          cc+=a[i];
  }
        if(cc==0)
            System.out.print("-1");
        else
            System.out.print(cc);
  }
  else
      System.out.print(a[0]);
 }
}
