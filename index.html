import java.util.*;
public class Hello {

 public static void main(String[] args) {
  Scanner x = new Scanner(System.in);
  String a = x.next();
  int t;
  for(int i=0;i<a.length();i++) {
   if(a.charAt(i)%2!=0){
   t= Character.getNumericValue(a.charAt(i))+1;
   if(t<10)
    System.out.print(t);
   else
    System.out.print(0);
   }
   else{
   t = Character.getNumericValue(a.charAt(i))+2;
   if(t<10)
    System.out.print(t);
   else
    System.out.print(0);
   }
  }
 }
}
